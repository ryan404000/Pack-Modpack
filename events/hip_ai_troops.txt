add_namespace = hip_ai_troops

country_event = {   #Japan need more troops against China
	id = hip_ai_troops.1
	title = hip_ai_troops.1.t
	desc = hip_ai_troops.1.d
	
	hidden = yes
	fire_only_once = yes
	
	trigger = {
		original_tag = JAP
		has_war_with = CHI 
		is_ai = yes
		has_army_size = {
			size < 101
			type = infantry
		}
	}
	
	immediate = {
		add_manpower = -1057500
		load_oob = JAP_ai_china
		add_equipment_to_stockpile = {
			type = convoy_1
			amount = 1000
			producer = JAP
		}
		
	}
	
	
	option = {
	
	}
}

country_event = {   #Manage soviet excess of light tanks

	fire_only_once = yes
	
	trigger = {
		original_tag = SOV
		is_ai = yes	
		has_equipment = {
			light_tank_equipment > 9999
		}
	}
	
	immediate = {
		add_equipment_to_stockpile = {
			type = light_tank_equipment
			amount = -10000

		}
		add_equipment_to_stockpile = {
			type = medium_tank_equipment
			amount = 7500
			producer = SOV
		}
	}
	
	
	option = {
	
	}
}

country_event = {   #Soviet need more troops
	id = hip_ai_troops.4
	title = hip_ai_troops.4.t
	desc = hip_ai_troops.4.d
	
	hidden = yes
	fire_only_once = yes
	
	trigger = {
		original_tag = SOV
		is_ai = yes	
		has_war_with = GER 
		GER = {
			is_ai = no
		}
		OR = {
			NOT = {
				controls_state = 242 #Smolensk
			}
			NOT = {
				controls_state = 221 #Karkov
			}
			NOT = {
				controls_state = 195 #Leningrad
			}
		}
	}
	
	immediate = {
		set_technology = {
			basic_light_td = 1
			basic_light_art = 1
			basic_light_spaa = 1
			basic_medium_tank = 1
			basic_medium_td = 1
			basic_medium_art = 1
			basic_medium_spaa = 1
			motorized_rocket_unit = 1
			mechanised_infantry = 1
			interwar_antiair = 1
		}
		
		add_ideas = {
			service_by_requirement
		}
		add_manpower = -1700000
		load_oob = SOV_ai_unit
	}
	
	
	option = {
	
	}
}
