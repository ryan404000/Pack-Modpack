country_event = {
	id = hip_italy.1
	title = hip_italy.1.t
	desc = hip_italy.1.d
	picture = GFX_report_event_generic_rally2

	fire_only_once = yes

	trigger = {
		tag = ITA		
		enemies_strength_ratio > 1
		has_war_with = USA
		NOT = {
			controls_state = 448 #tripoli
		}
		OR = {
			has_war_support < 0.6
			AND = {	
				casualties > 2500000
				is_ai = yes
			}
			AND = {	
				casualties > 1500000
				is_ai = no
			}			
		}
		OR = {
			any_country {
				is_in_faction_with = USA
				controls_state = 2			
			}
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe
				is_controlled_by = ENG
			}
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe
				is_controlled_by = CAN
			}			
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe				
				is_controlled_by = USA
			}
			any_owned_state = {
				is_core_of = ITA
				is_on_continent = europe				
				is_controlled_by = FRA
			}			
		}
	}
		
	mean_time_to_happen = {
		days = 0
	}	
	
	immediate = {
		kill_country_leader = yes
	}
	
	option = {
		name = hip_italy.1.a
		ai_chance = { factor = 100 }
		create_country_leader = {
			name = "Pietro Badoglio"
			desc = "POLITICS_PIETRO_BADOGLIO_DESC"
			picture = "Portrait_Italy_Pietro_Badoglio.dds"
			expire = "1965.1.1"
			ideology = conservatism
			traits = {
			}
		}
		set_politics = {
			ruling_party = democratic 
			elections_allowed = yes
		}
		add_popularity = {
			ideology = democratic 
			popularity = 0.35
		}
		add_popularity = {  
			ideology = communism
			popularity = 0.15
		}
		add_popularity = {
			ideology = fascism
			popularity = -0.10
		}			
		if = {
			limit = {
				ENG = { exists = yes }
				ENG = { is_faction_leader = yes }
			}			
			ENG = { country_event = { id = hip_italy.2 days = 1 } }
		}
		if = {
			limit = {
				USA = { exists = yes }
				USA = { is_faction_leader = yes }
			}			
			USA = { country_event = { id = hip_italy.2 days = 1 } }
		}		
	}
	option = {
		name = hip_italy.1.b
		ai_chance = { factor = 0 }
		start_civil_war = {
			ideology = communism
			size = 0.30
		}
		random_other_country = {
			limit = {
				original_tag = ITA
				has_government = communism
			}
			set_cosmetic_tag = ITA_SOV
			set_equipment_fraction = 0.3
			create_country_leader = {
				name = "Palmiro Togliatti"
				desc = "POLITICS_PALMIRO_TOGLIATTI_DESC"
				picture = "GFX_Portrait_Italy_Generic_1"
				expire = "1964.08.21"
				ideology = stalinism
			}			
		}		
	}	
}
